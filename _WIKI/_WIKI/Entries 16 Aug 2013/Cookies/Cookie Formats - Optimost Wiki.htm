// This function generates an "identifier space" that can contain up to 46,655 requests.
//
// Auto-generated IDs are a 12-character string divided into 8 fields:
//
// BIT  FIELD                   RANGE
//   z  Version                 1-35
//   z  Year                    1-35
//   z  Month                   1-12
//   z  Day                             1-31
//   z  Server ID               1-35
// zzz  Executor                1-46,655
// zzz  Request count   1-46,655
//   ?  Checksum                [0-9A-Z]
//
// The value for each field is an integer converted to BASE-36.
// This means that each executor can serve NO MORE THAN 46,655 requests and
// an individual "Server ID" can serve NO MORE THAN (46,655 * 46,655) requests in a day.
//
// Individual identifiers are created by concatenating the generated space prefix with
// the request index and checksum.
//
function HitCreateIDSpace($nowtime, $serverId, $executorId)
{
        $year = gmdate('Y', $nowtime) - 2004;

        if ($year > 30)
                LogError('Year identifier value is nearing encoding limit');
        if ($serverId > 30)
                LogError('Server identifier value is nearing encoding limit');
        if ($executorId > 40000)
                LogError('Executor identifier value is nearing encoding limit');

        return implode(, array(
                HitEncodeIDField(2),                                                            //   z 1-35 (2=OM_HITIDSPACE_BY)
                HitEncodeIDField($year),                                                        //   z 2004-2040
                HitEncodeIDField(gmdate('n', $nowtime)),                        //   z 1-12
                HitEncodeIDField(gmdate('j', $nowtime)),                        //   z 1-31
                HitEncodeIDField($serverId),                                            //   z 1-35
                HitEncodeIDField($executorId, 3),                                       // zzz 1-46,655
                ));
}



// Creates a session identifier based on the request context.
// Session will be 25 or 37 characters in length.
//
// z            Format version
// zzz          SegmentID (46,655 max)
// zzz          WaveID (46,655 max)
// zzz          CreativeID (46,655 max)
// zzzzzzzzzzzz VisitorID
// zzzzzzzzzzzz ImpressionID (not included if same as VisitorID)
// xxx          Checksum (partial MD5)
//
function HitCreateSessionIdentifier($segmentid, $waveid, $creativeid, $visitorid, $impressionid)
{
        if (!$segmentid)
                return "$waveid,0,$creativeid,$impressionid"; // Pre-2006 session identifier

        if ($segmentid > 40000)
                LogError('Persona identifier value is nearing encoding limit');
        if ($waveid > 40000)
                LogError('Wave identifier value is nearing encoding limit');
        if ($creativeid > 40000)
                LogError('Creative identifier value is nearing encoding limit');
        if (strlen($visitorid) != 12)
                LogError("Visitor identifier value is beyond encoding limit ($visitorid)");
        if (strlen($impressionid) != 12)
                LogError("Impression identifier value is beyond encoding limit ($impressionid)");

        $sessionId =
                'a' .
                HitEncodeIDField($segmentid, 3) .
                HitEncodeIDField($waveid, 3) .
                HitEncodeIDField($creativeid, 3) .
                $visitorid .
                (($impressionid != $visitorid) ? $impressionid : );

        return $sessionId.substr(md5($sessionId.'most'), -3);
}
