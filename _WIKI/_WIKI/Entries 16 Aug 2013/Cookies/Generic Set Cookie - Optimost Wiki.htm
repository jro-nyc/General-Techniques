opXMLSetCookieA

This sets the cookie with a name, value, domain and expiry time. 

function opXMLSetCookieA(n, v, d, e){var de = new Date;de.setTime(de.getTime() + e * 1000);document.cookie = n + "=" + escape(v) + ((e==null) ? "" : ("; expires=" + de.toGMTString())) + "; path=/" + ((d==null) ? "" : (";domain=" + d));};

opXMLGetDocumentSLD

This gets the domain. Note that this will not work for domains such as google.co.uk - it just returns ".co.uk". 

function opXMLGetDocumentSLD(){var sld = document.domain;var dp = sld.split(".");var l = dp.length;if (l < 2) sld = null;else if (!isNaN(dp[l-1]) && !isNaN(dp[l-2])) sld = null;else sld = "." + dp[l-2] + "." + dp[l-1];return sld;}

setTheSticky

function setTheSticky(name,value,xtime){
	if(xtime=='null') xtime=null
	else {
		xtime=parseFloat(xtime)*60;
	}
	opXMLSetCookieA(name, value, opXMLGetDocumentSLD(), xtime);
}
