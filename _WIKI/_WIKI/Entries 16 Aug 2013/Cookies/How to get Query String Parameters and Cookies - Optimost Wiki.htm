Get Query String Parameters

This function gets all the query string parameters and puts them in an object (which imitates an associative array). 

function getParams(){
	var query  = unescape(top.location.search.substring(1)).split('&');
	var params = new Object();

	for (i = 0; i < query.length; i++)	{
		var tmp = query[i].split('=');
		params[tmp[0]] = tmp[1];
	}
	if(query.length>0) return params;
	else return null;
}

If you wanted to get the value of a query string parameter called "thisParam", you would do this: 

allParams = getParams();
thisParamValue=allParams['thisParam'];

Rememeber to check this value exists before using it...

Set a Cookie

function setDaCookies(cookieName,cookieValue){
	document.cookie= cookieName + "=" + escape(cookieValue) + "; path=/;";	
}

Get a Specific Cookie

function getDaCookies(cookieValue){
var _opcookies = new Object;
	if (document.cookie.length > 0) { var _ca = document.cookie.split("\;");
	for (var i=0; i < _ca.length; i++) { var _cv = _ca[i].split("=");
	while (_cv[0].substring(0, 1) == " ") _cv[0] = _cv[0].substring(1, _cv[0].length);
	_opcookies[_cv[0]] = unescape(_cv[1]); } }
	if(_opcookies[cookieValue] != null) {return _opcookies[cookieValue];}
	else {return null;} 	
}
