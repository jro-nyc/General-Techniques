Useful functions for DOM templates

In doing DOM manipulation, here are some functions I found to be very useful, and you might, too. The names I use can be changed if you'd like, just understand what the functions do and how they do it.

opCleanNodeAddText(obj,innerVal)

If you want to replace the contents of a node with a text string - and that could include tags inside of it - just pass this function the node you want to do this to, and the string. 

function opCleanNodeAddText(obj,innerVal){
	if(obj!=null){
		children = obj.childNodes;
		for(var i=children.length-1;i>-1;i--){
			obj.removeChild(children[i]);       
		} 
		newText=document.createTextNode(innerVal);
		obj.appendChild(newText);    
	}
}

opCleanNodeAddText_Node(obj,innerVal,container,styles)

A slight mod over function above - can still add a text node if the last two parameters are null, but if you want the text inside some new container, then add that tag name as a string, and you can even pass in a style for the new container.
Also added it to the $opt object as a function. 

$opt={};
$opt.opCleanNodeAddText_Node=function(obj,innerVal,container,styles){
    if(obj!=null){
	children = obj.childNodes;
        for(var i=children.length-1;i>-1;i--){
                obj.removeChild(children[i]);       
        }
        if(container==null || container=="text"){
            newText=document.createTextNode(innerVal);
            obj.appendChild(newText);
        }
        else if(typeof(container)=="string"){
            newNode=document.createElement(container);
            if(newNode){
                if(typeof(styles)=="string")newNode.style.cssText=styles;
                newText=document.createTextNode(innerVal);
                newNode.appendChild(newText);
                obj.appendChild(newNode);                                    
            }                    
        }
    }
}

opReplaceImageNode(origNode,newSrc,type,widthPX,heightPX)

If you want to replace an image, there is an inherent issue with replacing the src URL - if it takes a while to load that image, you will see the original image flicker to the new one. The resolution is to just replace it. Here we start with the original image node, pass in the new URL and size. Note that:

  • You can replace a regular image, OR an input image. The type parameter should specify "input" if it is an input object. A null or any other value defaults to a "img" object.
  • If you pass no size, none will be set. Send the size with px at the end, as in 100px for 100 pixels.
  • It returns the node in case you want to do other things to it, IE add a class, add onmouseover rolloves, onclick functionality, etc 
function opReplaceImageNode(origNode,newSrc,type,widthPX,heightPX){//This can be a regular image or an input image
	if(origNode!=null && newSrc!=null){
		if(type==null || type.toLowerCase()!='input') type='img';
		var newNode=document.createElement(type);
		if( type.toLowerCase()=='input')newNode.type='image';
		newNode.src=newSrc;
		if(widthPX!=null && widthPX.indexOf('px')!=-1)newNode.style.width=widthPX;
		if(heightPX!=null && heightPX.indexOf('px')!=-1)newNode.style.height=heightPX;		
		origNode.parentNode.replaceChild(newNode,origNode);
		return newNode;
	}
	else return null;
}

FOR NAMED INPUTS — opReplaceImageNode(origNode,newSrc,name,type,widthPX,heightPX)

Like above, but for form images that have names. Here you have to account for the IE bug that you cannot add a name except when creating the element. 

function opReplaceImageNode(origNode,newSrc,name,type,widthPX,heightPX){//This can be a regular image or an input image
	if(origNode!=null && newSrc!=null){
		if(type==null || type.toLowerCase()!='input') type='img';
		else if(type.toLowerCase()=='input')type='image';
		if(name==null ||(name!=null && name=='')){
			try {//IE has issues with making inputs with names, do them like this
				newNode=document.createElement("<input type='"+type+"' name='"+name+"'/>");  
			} 
			catch (ex) {
				newNode = document.createElement("input");
				newNode.type = type;
				newNode.name = name;  
			}
		}
		else {
			newNode = document.createElement("input");
			newNode.type = type;
		}
		newNode.src=newSrc;
		if(widthPX!=null && widthPX.indexOf('px')!=-1)newNode.style.width=widthPX;
		if(heightPX!=null && heightPX.indexOf('px')!=-1)newNode.style.height=heightPX;		
		origNode.parentNode.replaceChild(newNode,origNode);
		return newNode;
	}
	else return null;
}

findParentNode(obj, type,nestingLevel)

Explained in Template Example, this gets a parent node above you based on the tag. 

function findParentNode(obj, type,nestingLevel){
	var newNode=obj;
	if(newNode!=null){
		var finished=false;
		var nesting=0;
		var num=0;
		do{			
			if(newNode.parentNode==null) return null; //If no parent Node, outside top DOM element, done
			newNode=newNode.parentNode; //Find our parent Node, set it as the new node
			num++;
			if(newNode.tagName==null) return null; //if no Tag Name, the document node, too high, exit
			else if(newNode.tagName.toLowerCase()==type.toLowerCase()){//If we have a type match...
				nesting++;
				if(nesting==nestingLevel) finished=true; //and if we are at the right nesting level, FOUND
			}
		}while(newNode.tagName.toLowerCase()!=type.toLowerCase()||!finished);	//Wrong type and not Found, loop again
		return(newNode);
	}
	else return null;
}

opt.getElementsByClassName(startObj,clsName)

Explained in Template Example, this gets all the nodes that have a matching class, no matter what the node types are. it is a function of the opt object since Firefox 3 has a getElementsByClassName function. 

opt=new Object;
opt.getElementsByClassName = function(startObj,clsName){
	var retVal = new Array();
	var elements = startObj.getElementsByTagName("*");
	for(var i = 0;i < elements.length;i++){
		if(elements[i].className.indexOf(" ") >= 0){
			var classes = elements[i].className.split(" ");
			for(var j = 0;j < classes.length;j++){
				if(classes[j] == clsName)
				retVal.push(elements[i]);
			}
		}
		else if(elements[i].className == clsName)
			retVal.push(elements[i]);
	}
	return retVal;
}

findNode_BasedOnTagAndContent

Very useful. 

function findNode_BasedOnTagAndContent(startObj,tagName,attrName,matchValue){
	foundNode=null;
	tags=startObj.getElementsByTagName(tagName);
	for(var j=0;j<tags.length;j++){
		if(tags[j][attrName].indexOf(matchValue)!=-1){
			foundNode=tags[j];
		}
	}	
	return foundNode;
}

opGetTextNode

This pulls out a text node from a container based on matching text. Text nodes can be a pain to find and you will get different results in IE and FF - in FF, you get a lot more text nodes than in IE. 

function opGetTextNode(element, text){
	if(!element) return null;
	for(var i=0; i<element.childNodes.length; i++)	{
		if(element.childNodes[i].nodeValue)		{
			if(text == "" || (element.childNodes[i].nodeValue.toLowerCase().indexOf(text.toLowerCase()) > -1)){
				return element.childNodes[i];
			}
		}
	}
	return null;
}
