Code to track submits

First, the following code goes in the setup block. 

<script language="javascript" type="text/javascript">
opt=new Object;
opt.getElementsByClassName = function(startObj,clsName){
	var retVal = new Array();
	var elements = startObj.getElementsByTagName("*");
	for(var i = 0;i < elements.length;i++){
		if(elements[i].className.indexOf(" ") >= 0){
			var classes = elements[i].className.split(" ");
			for(var j = 0;j < classes.length;j++){
				if(classes[j] == clsName)
				retVal.push(elements[i]);
			}
		}
		else if(elements[i].className == clsName)
			retVal.push(elements[i]);
	}
	return retVal;
}
opt.absoluteAddr=function(theURL) {
	if (theURL.substr(0,4) != "http"){
		if(document.getElementsByTagName('base')[0] != null){ // If we have a base statement, use that to set the root path
			theRoot = document.getElementsByTagName('base')[0].href;
		}
		else{  // if no base statement                 
			theRoot = window.location + "";
		}
			
		var orig = theRoot.split("//");
		protocol = orig[0] + '//';
		var dir = orig[1].split("/");
		var dirs = "";
		
		for(var i=0;i<dir.length-1;i++){
			dirs += (dir[i]!=''?dir[i] + '/':''); 
		}
			
		theBaseDomain = protocol + dir[0];
		theDomain = protocol + dir[0] + '/';
		theRoot = protocol + dirs;
			
		if (theURL.substr(0,1) == "/"){ // relative to document domain
			theURL = theBaseDomain + theURL;
		}
		else {
			theURL = theRoot + theURL;
		}
	}
	return theURL;
}
opt.clickthru = function (o, ct, qsParam) {
	if (typeof(o.originaltarget) != 'undefined') return false; //see if we have done this before
		
	o.originaltarget = 'done'; // Used in tandem with line above to prevent from  re-instrumenting the target object
	if (o.tagName.toLowerCase() == "form") h=opt.absoluteAddr(o.action);
	else h=opt.absoluteAddr(o.href);   
		
	o.originaltarget = h;
	newH = ct+(ct.indexOf('?')==-1?'?':'&') + qsParam +'=' + escape(h);
	if (o.tagName.toLowerCase() == "form"){
		if(o.method.toLowerCase()=='post'){//Adding for POST and href is same				
			o.action= newH;
		}
		else {//get Click, need to add Clickthru as action and form post value as a hidden input
			var newInput=document.createElement('input');
			newInput.setAttribute('type','hidden');
			newInput.setAttribute('name',qsParam);
			newInput.setAttribute('value',h);	
			o.appendChild(newInput);
			o.action= ct;			
		}
	}
	else { //Adding for POST and href is same
		o.href= newH;
	}
	return true;
}
</script>

Then the following code goes in the body of the page. 


<script language="javascript" type="text/javascript">
theClick='{{$clickthru.form1.url}}';
allForms=document.forms
for(j=0;j<allForms.length;j++){
	if(allForms[j].onsubmit==null){	
		allForms[j].onsubmit=new Function('opt.clickthru(this,"'+ theClick +'", "oppost");');
	}
	else{
		tempOnClick=allForms[j].onsubmit.toString();
		tempOnClick=tempOnClick.substring(tempOnClick.indexOf("{") + 1, tempOnClick.lastIndexOf("}"));
		allForms[j].onsubmit=new Function('opt.clickthru(this, "'+ theClick +'", "oppost");'+tempOnClick);		
	}			
}
</script>
