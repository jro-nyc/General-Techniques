Parsing through the node in a containter

It's fine when we have to parse through a container's child nodes to find a node of interest. But what if the nodes you want are a mix of "child nodes" of the cotainer and "child nodes" of the "child nodes" or even "child nodes" of the "child nodes" of the "child nodes"? What you need is recursive code.

Our Scenario

A customer decides they want to highlight all the numbers in a block by changing them from black to red text. In order to do this, we need to:

  1. Write a regexp based matching and replacing function to find the numbers, where they start and end.
  2. Write another function that, from a starting container node, find all the text nodes, and run each text node through this and, if the text node has numbers, replace the numbers in the text node with <span> tags with a style containing the new color.
  3. Within each container, you must start at the last text node and work backwards. This is because:
    1. Adding the span tags will change the number of nodes
    2. If you start at the end, the number of text nodes changes for the content after this node
    3. But since we start with the last node, there will be no nodes of interest after this one
  4. If you find element nodes within the container, use these element nodes as the new "container" and recursively call your function.

walkTheDOM(startRegexp,endRegexp,container,styleChange)

I wrote this myself for this example. First, we have the recursive function: 

function walkTheDOM(startRegexp,endRegexp,container,styleChange) {
	var TEXT_NODE=3;
	var children=container.childNodes; 
	for(var i=children.length-1;i>-1;i--){
		if(children[i].nodeType==TEXT_NODE) {
			changePatternTextNode(startRegexp,endRegexp,children[i],styleChange);
		}
		else walkTheDOM(startRegexp,endRegexp,children[i],styleChange); 	
	} 
}

Where I pass in:

  • startRegexp - a regexp representation of the starting match
  • endRegexp - a regexp representation of the ending match - can be a null if you are only interested in matching startRegexp
  • container - our starting container
  • styleChange - the style change we want to implement

So all I do here is:

  1. I get all the childNodes in this container and starting at the last childNode,
  2. I check to see if its a text node
    1. If its a text node, I execute my matching and replacing function
    2. If its not a text node, I use that as my container node and call this same function recursively

function changePatternTextNode(startRegexp,endRegexp,container,styleChange){
	var rxpSt=new RegExp(startRegexp); 
	var rxpEnd=(endRegexp!=null?new RegExp(endRegexp):endRegexp); //if endRegexp=null, don't create RegExp object
	end=[];
	start=[];	
	var offset=0;
	match=true;
	do {  //Since we may have multiple matches in any particular text node, unless endRegexp is null
		var val=container.nodeValue.substring(offset,container.nodeValue.length);
		if(rxpSt.exec(val)){ //if the nodeValue contains the start match.....
			rVal=val.substring(rxpSt.exec(val).index,val.length);
			if(rxpEnd==null){//If endRegexp is null, we have a match - only care about "startRegexp"
				start.push(rxpSt.exec(val).index+offset);
				end.push(rxpSt.exec(val).index+startRegexp.toString().length+offset);
				offset+=rxpSt.exec(val).index+startRegexp.toString().length;		
				match=false; //Won't have multiple matches for this case
			}
			else if(rxpEnd.exec(rVal)){ //if the nodeValue contains the end match after it....
				start.push(rxpSt.exec(val).index+offset);	
				end.push(rxpSt.exec(val).index+rxpEnd.exec(rVal).index+offset);
				offset+=rxpSt.exec(val).index+rxpEnd.exec(rVal).index;
			}
			else match=false;
		}
		else match=false;	
	} while(match);
	if(start.length>0){
		var myNode=container.nodeValue;	//Get the value you will manipulate
		nodeBucket=[];	
		for(var j=start.length-1;j>-1;j--){		
			if(j<start.length-1) var newText3=document.createTextNode(myNode.substring(end[j],start[j+1]));	
			else var newText3=document.createTextNode(myNode.substring(end[j],myNode.length));	
			nodeBucket.push(newText3);		

			var newSpan=document.createElement('span');	
			newSpan.style.cssText=styleChange;
			var newText2=document.createTextNode(myNode.substring(start[j],end[j]));		
			newSpan.appendChild(newText2);
			nodeBucket.push(newSpan);		
		}			
		var newText=document.createTextNode(myNode.substring(0,start[0]));			
		nodeBucket.push(newText);		
		for(var j=0;j<nodeBucket.length;j++){		
			container.parentNode.insertBefore(nodeBucket[j],container.nextSibling);
		}
		container.parentNode.removeChild(container);	
	}
}

I pass in the same parameters as the walkTheDOm function, and I:

  1. Get the nodeValue for this text node.
  2. Run it through a do{..}while that looks for matches of my starting and ending patterns.
    1. Saves the start and end match locations in separate arrays
    2. Continues doing this until there are no more matches
    3. If the ending pattern passed in endRegexp is null, that means was are looking for a match of startRegexp only, and we do not need to look for more matches within this node once startRegexp matches.
  3. Then, if there were matches for this text node, starting at the last match, we make the necessary changes.

function changePatternTextNode(startRegexp,endRegexp,container,styleChange){
	var rxpSt=new RegExp(startRegexp); 
	var rxpEnd=new RegExp(endRegexp); 
	end=[];
	start=[];	
	var offset=0;
	match=true;
	do {  //Since we may have multiple matches in any particular text node
		var val=container.nodeValue.substring(offset,container.nodeValue.length);
		if(rxpSt.exec(val)){ //if the nodeValue contains the start match.....
			rVal=val.substring(rxpSt.exec(val).index,val.length);
			if(rxpEnd.exec(rVal)){ //and the nodeValue contains the end match after it....
				start.push(rxpSt.exec(val).index+offset);	
				end.push(rxpSt.exec(val).index+rxpEnd.exec(rVal).index+offset);
				offset+=rxpSt.exec(val).index+rxpEnd.exec(rVal).index;
			}
			else match=false;
		}
		else match=false;	
	} while(match);
	if(start.length>0){
		var origNode=container.nodeValue;
		var myNode=container.nodeValue;
		for(var j=start.length-1;j>-1;j--){					
			myNode=myNode.substring(0,start[j])+
			'<span style="'+styleChange+'">'+myNode.substring(start[j],end[j])+'</span>'+
			myNode.substring(end[j],myNode.length);													
		}

		newSpan.innerHTML=myNode;

		container.parentNode.removeChild(container);
	}
}
