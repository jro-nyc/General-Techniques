Pull Out the Root Path

Code below pulls out root path for converting relative path links to full path
if(document.getElementsByTagName('base')[0] != null){ // If we have a base statement, use that to set the root path
	daRoot = document.getElementsByTagName('base')[0].href;
	var orig = daRoot.split("//");
	protocol = orig[0] + '//';
	var dir = orig[1].split("/");
	var dirs = "";
	for(var i=0;i<dir.length-1;i++){
		dirs += (dir[i]!=''?dir[i] + '/':'');  
	
	}
	daBaseDomain = protocol + dir[0];
	daDomain = protocol + dir[0] + '/';
	daRoot = protocol + dirs;
}
else {
	var sPage = window.location.pathname;
	var dir = sPage.split("/");
	var dirs = "";
	var baseURL;
	for(var i=0;i<dir.length-1;i++){
		dirs += (dir[i]!=''?dir[i] + '/':''); 
	
	}
	daBaseDomain = location.protocol+"//"  + location.host;
	daDomain = location.protocol+"//"  + location.host + '/';
	daRoot = location.protocol+"//" + location.host + "/" + dirs;
}

Pull out the Form action

//Code here pulls out the form post action from the live page and stores it
formName = 'frm';
formObj = document.forms[formName];
var actObj = formObj.attributes.getNamedItem('action');
if(actObj!=null)origAction=actObj.nodeValue;
else origAction='';

daUrlNoQS = origAction.split('?')[0]; // for the case when http is a Query string paramter
daUrlNoQS = daUrlNoQS.substr(0,9);
if(daUrlNoQS.substr(0,1)=='/') 	daRoot=daBaseDomain; //If the action has a "/" as the first character
if(origAction=='') {
	newPostLink= document.location;
}	
else {
	newPostLink = ((daUrlNoQS.indexOf('http')==-1)?daRoot:'') + origAction; 
}

Pull out and store form elements

//Here, we are pulling the hidden variables (whose values may be dynamic) from the first Form, which has no name 
//Here, we are pulling the hidden variables (whose values may be dynamic) from the second Form - ACMEForm - along with 
//regular inputs
formObj = document.forms[formName];

vals='';
hiddenList='';
inputVals = new Object;
for (i=0;i<formObj.length;i++){
	if(formObj.elements[i].type.toLowerCase()== 'hidden') {
		hiddenList+='<input type="hidden" name="'+formObj.elements[i].name+'" '+ 
		(formObj.elements[i].id!=null?(formObj.elements[i].id!=''?'id="'+formObj.elements[i].id+'" ':''):'')+
		' value="'+formObj.elements[i].value+'" />\n';
	}
	else if(formObj.elements[i].type.toLowerCase()== 'checkbox'){
		inputVals[formObj.elements[i].name]=(formObj.elements[i].checked==true?'true':'false');
	}
	else if(formObj.elements[i].type.toLowerCase()== 'radio'){
		if(formObj.elements[i].checked==true) inputVals[formObj.elements[i].name]=formObj.elements[i].value;
		//alert(formObj.elements[i].name+' is a radio button and is '+(formObj.elements[i].checked?'CHECKED':'NOT checked')+' and has a value of '+inputVals[formObj.elements[i].name]);
	}
	else{
		inputVals[formObj.elements[i].name]=formObj.elements[i].value;
		vals+=formObj.elements[i].name+' has a value of '+formObj.elements[i].value+'\n';
	}	
}

Add Back the Original Values

formObj = document.forms[formName];
daPost = '{{$clickthru.Form1.url}}'; 
//This code below add backs the form values, if necessary
for (i=0;i<formObj.length;i++) {
	if(inputVals[formObj.elements[i].name]!=null){	
		n=formObj.elements[i].name;
		if(formObj.elements[i].type=='checkbox'){
			if(inputVals[n]=='true')formObj.elements[i].checked = true;
		}
		else if(formObj.elements[i].type=='radio'){
			if(formObj.elements[i].value == inputVals[n]) {
				formObj.elements[i].checked = true;
			}
		}
		else {
			formObj.elements[i].value=inputVals[n];
		}	
	}
}
